1. Context

A random-intercept binomial GLMM models a binary outcome \(y_{ij} \in \{0,1\}\) clustered within group \(i\):

\[ \operatorname{logit}\{\Pr(y_{ij}=1 \mid b_i)\} = x_{ij}^{\top}\beta + b_i,\qquad b_i \sim N(0,\sigma_b^2). \]

ds.vertGLMM(method = "pql_aggregate") uses a PQL working-response update and solves the random-intercept weighted LMM from aggregate sufficient statistics. The route is deliberately Ring127-only because the working statistics require repeated share-domain products and reciprocals.

The validation below runs the same deterministic dataset as a central lme4::glmer reference and as two DSLite vertical partitions:

Mode Split
K = 2 s1: x1, x2; s2: cluster_id, x3, y_bin
K = 3 s1: x1; s2: x2; s3: cluster_id, x3, y_bin

Non-disclosure invariants

Invariant Statement
D-INV-1 Per-patient covariates never leave their origin server.
D-INV-2 Fitted probabilities, working responses, PQL weights, residuals and row scores remain server-side or in Ring127 additive shares.
D-INV-3 The client receives low-dimensional coefficients, variance components, guarded cluster-level PQL sufficient statistics and cluster-level BLUPs, not patient-level vectors.
D-INV-4 Original cluster labels are not returned; clusters below datashield.privacyLevel fail closed.
D-INV-5 The returned-object audit flags any atomic vector of length \(n\); only cluster-length b_hat and cluster_sizes are accepted.

build_pooled <- function() {
  set.seed(20260502L)
  n_cluster <- 10L
  cluster_size <- 5L
  n <- n_cluster * cluster_size
  cluster_id <- rep(seq_len(n_cluster), each = cluster_size)
  x1 <- stats::rnorm(n)
  x2 <- stats::rnorm(n)
  x3 <- stats::rnorm(n)
  b_i <- stats::rnorm(n_cluster, sd = 0.65)
  eta <- -0.35 + 0.55 * x1 - 0.45 * x2 + 0.35 * x3 + b_i[cluster_id]
  p <- stats::plogis(eta)
  y_bin <- stats::rbinom(n, 1L, p)
  data.frame(
    patient_id = sprintf("GLMM%04d", seq_len(n)),
    cluster_id = cluster_id,
    x1 = x1,
    x2 = x2,
    x3 = x3,
    y_bin = y_bin)
}

split_tables <- function(pooled, K) {
  if (K == 2L) {
    list(
      s1 = pooled[, c("patient_id", "x1", "x2")],
      s2 = pooled[, c("patient_id", "cluster_id", "x3", "y_bin")]
    )
  } else if (K == 3L) {
    list(
      s1 = pooled[, c("patient_id", "x1")],
      s2 = pooled[, c("patient_id", "x2")],
      s3 = pooled[, c("patient_id", "cluster_id", "x3", "y_bin")]
    )
  } else {
    stop("This vignette validates K = 2 and K = 3 only.")
  }
}

connect_dslite <- function(tables, symbol = "D") {
  server <- DSLite::newDSLiteServer(
    tables = tables,
    config = DSLite::defaultDSConfiguration(include = c("dsBase", "dsVert")))
  dslite.server <<- server
  builder <- DSI::newDSLoginBuilder()
  for (nm in names(tables)) {
    builder$append(server = nm, url = "dslite.server",
                   table = nm, driver = "DSLiteDriver")
  }
  DSI::datashield.login(builder$build(), assign = TRUE,
                         symbol = symbol, opts = list(server = server))
}

4. Federated PQL Fits

compare_named <- function(a, b) {
  nm <- intersect(names(a), names(b))
  delta <- as.numeric(a[nm]) - as.numeric(b[nm])
  data.frame(
    coef = nm,
    dsVert = as.numeric(a[nm]),
    glmer = as.numeric(b[nm]),
    delta = delta,
    abs_delta = abs(delta),
    row.names = NULL)
}

audit_returned_object <- function(x, n, n_clusters) {
  bad <- character()
  cluster_level <- character()
  walk <- function(obj, path) {
    if (is.null(obj) || is.environment(obj) || is.function(obj) ||
        inherits(obj, "formula") || inherits(obj, "call")) {
      return(invisible(NULL))
    }
    if (is.atomic(obj)) {
      len <- length(obj)
      if (len == n) bad <<- c(bad, path)
      if (len == n_clusters && len != n) {
        cluster_level <<- c(cluster_level, path)
      }
      return(invisible(NULL))
    }
    if (is.matrix(obj) || is.data.frame(obj)) {
      d <- dim(obj)
      if (length(d) == 2L && any(d == n)) bad <<- c(bad, path)
      if (length(d) == 2L && any(d == n_clusters) && !any(d == n)) {
        cluster_level <<- c(cluster_level, path)
      }
    }
    if (is.list(obj)) {
      nm <- names(obj)
      if (is.null(nm)) nm <- as.character(seq_along(obj))
      for (i in seq_along(obj)) walk(obj[[i]], paste0(path, "$", nm[[i]]))
    }
    invisible(NULL)
  }
  walk(x, "fit_ds")
  list(pass = length(unique(bad)) == 0L,
       disallowed_length_n_paths = unique(bad),
       cluster_level_paths = unique(cluster_level),
       n = n,
       n_clusters = n_clusters)
}

fit_one <- function(K) {
  options(dsvert.require_trusted_peers = FALSE,
          datashield.privacyLevel = 5L,
          datashield.errors.print = TRUE)
  conns <- connect_dslite(split_tables(pooled, K))
  on.exit(try(DSI::datashield.logout(conns), silent = TRUE), add = TRUE)
  psi <- ds.psiAlign("D", "patient_id", "DA",
                      datasources = conns, verbose = FALSE)
  t0 <- proc.time()[3]
  fit_ds <- ds.vertGLMM(
    formula = formula,
    data = "DA",
    cluster_col = "cluster_id",
    max_outer = 1L,
    inner_iter = 2L,
    tol = 1e-2,
    method = "pql_aggregate",
    ring = 127L,
    compute_se = FALSE,
    verbose = FALSE,
    datasources = conns)
  runtime_s <- unname(proc.time()[3] - t0)
  cmp <- compare_named(fit_ds$coefficients, ref_beta)
  list(
    K = K,
    n = nrow(pooled),
    n_matched = psi[[1]]$n_matched,
    runtime_s = runtime_s,
    coefficients = cmp,
    max_abs_delta = max(cmp$abs_delta),
    sigma_b2_ds = fit_ds$sigma_b2,
    sigma_b2_ref = ref_sigma_b2,
    sigma_b2_abs_delta = abs(fit_ds$sigma_b2 - ref_sigma_b2),
    audit = audit_returned_object(fit_ds, nrow(pooled),
                                  length(unique(pooled$cluster_id))),
    diagnostics = list(iterations = fit_ds$iterations,
                       phi = fit_ds$phi,
                       tau = fit_ds$tau,
                       trace = fit_ds$trace))
}

The audited aggregate-PQL route matches the centralized glmer singular fixture with maximum fixed-effect gaps of 3.756e-04 and random-effect variance gaps of 1.008e-12 for both K modes. The returned-object audit passes in K = 2 and K = 3; no patient-length atomic vector is visible to the analyst.
