dsVert Validation: Privacy-Preserving Vertical GLM, Correlation, and PCA

This vignette validates dsVert against centralised R on two public clinical datasets:

  • birthwt (MASS package): low birth weight risk factors, \(K=2\) servers
  • CGD (survival package): chronic granulomatous disease trial, \(K=3\) servers

We compare coefficients, standard errors, p-values, and canonical deviance for all three GLM families (Gaussian, binomial, Poisson), plus Pearson correlation and PCA.

1. Data preparation

1.1 birthwt (\(K=2\))

The birthwt dataset contains 189 observations of risk factors for low birth weight at Baystate Medical Center. We split it vertically across two hospitals:

  • Hospital A: maternal demographics (age, weight, race, smoking)
  • Hospital B: clinical history and outcomes (preterm labour, hypertension, uterine irritability, physician visits, birth weight, low birth weight flag)
library(MASS)
data(birthwt)

df <- birthwt
df$patient_id <- sprintf("BW%03d", seq_len(nrow(df)))

set.seed(123)
core <- sample(df$patient_id, 160)
remaining <- setdiff(df$patient_id, core)
ids1 <- sample(c(core, sample(remaining, 15)))
ids2 <- sample(c(core, sample(remaining, 10)))

# Hospital A: maternal demographics
s1 <- df[df$patient_id %in% ids1, c("patient_id", "age", "lwt", "race", "smoke")]
# Hospital B: clinical + outcomes
s2 <- df[df$patient_id %in% ids2, c("patient_id", "ptl", "ht", "ui", "ftv", "bwt", "low")]

# Shuffle row order (PSI must handle this)
s1 <- s1[sample(nrow(s1)), ]
s2 <- s2[sample(nrow(s2)), ]

1.2 Centralised ground truth (birthwt)

# Merge as if data were pooled (ground truth)
merged <- merge(s1, s2, by = "patient_id")

gt_gaussian <- glm(bwt ~ age + lwt + race + smoke + ptl + ht + ui,
                   data = merged, family = gaussian())
gt_binomial <- glm(low ~ age + lwt + race + smoke + ptl + ht + ui,
                   data = merged, family = binomial())
gt_poisson  <- glm(ftv ~ age + lwt + race + smoke + ptl + ht + ui,
                   data = merged, family = poisson())

summary(gt_binomial)

1.3 CGD (\(K=3\))

The CGD dataset contains 128 patients from a clinical trial of interferon gamma for chronic granulomatous disease (Fleming & Harrington, 1991). We split vertically:

  • Hospital A: demographics (age, sex, height, weight)
  • Hospital B: treatment assignment (treatment arm, steroid use)
  • Hospital C: outcomes (infection status, total infection count)
library(survival)
data(cgd)

# One row per patient
d <- cgd[cgd$enum == 1, ]
d$n_infections <- as.integer(tapply(cgd$status, cgd$id, sum)[as.character(d$id)])
d$patient_id <- sprintf("CGD%03d", seq_len(nrow(d)))
d$treat_num <- as.integer(d$treat == "rIFN-g")
d$sex_num <- as.integer(d$sex == "male")

# All 128 patients on all servers (different variables)
s1 <- d[, c("patient_id", "age", "sex_num", "height", "weight")]
s2 <- d[, c("patient_id", "treat_num", "steroids")]
s3 <- d[, c("patient_id", "status", "n_infections")]

# Shuffle independently
set.seed(42); s1 <- s1[sample(nrow(s1)), ]
set.seed(43); s2 <- s2[sample(nrow(s2)), ]
set.seed(44); s3 <- s3[sample(nrow(s3)), ]

1.4 Centralised ground truth (CGD)

gt_cgd_gaussian <- glm(weight ~ age + sex_num + height + treat_num + steroids,
                       data = d, family = gaussian())
gt_cgd_binomial <- glm(status ~ age + sex_num + height + weight + treat_num + steroids,
                       data = d, family = binomial())
gt_cgd_poisson  <- glm(n_infections ~ age + sex_num + height + weight + treat_num + steroids,
                       data = d, family = poisson())

summary(gt_cgd_binomial)

2. Upload to Opal servers

Each server fragment is uploaded to its own Opal/Rock instance. In a real deployment, each institution would hold its own data; here we simulate this with local containers.

library(opalr)

# birthwt K=2
for (i in 1:2) {
  o <- opal.login("administrator", "password",
    url = sprintf("https://server%d.example.org", i))
  opal.table_save(o, list(s1, s2)[[i]], "bw_project", "bw_data",
                  overwrite = TRUE, force = TRUE)
  opal.logout(o)
}

# CGD K=3
for (i in 1:3) {
  o <- opal.login("administrator", "password",
    url = sprintf("https://server%d.example.org", i))
  opal.table_save(o, list(s1, s2, s3)[[i]], "cgd_project", "cgd_data",
                  overwrite = TRUE, force = TRUE)
  opal.logout(o)
}

3. dsVert validation: birthwt (\(K=2\))

3.1 Connect and align

library(DSI)
library(DSOpal)
library(dsVertClient)

builder <- DSI::newDSLoginBuilder()
builder$append(server = "hospitalA", url = "https://server1.example.org",
               user = "analyst", password = "...", table = "bw_project.bw_data")
builder$append(server = "hospitalB", url = "https://server2.example.org",
               user = "analyst", password = "...", table = "bw_project.bw_data")
conns <- DSI::datashield.login(logins = builder$build())
DSI::datashield.assign.table(conns, "D", "bw_project.bw_data")

# Align records — NAs are automatically removed before PSI
psi <- ds.psiAlign("D", "patient_id", "DA", datasources = conns)

3.2 Gaussian GLM (birth weight)

result <- ds.vertGLM(bwt ~ age + lwt + race + smoke + ptl + ht + ui,
                     data = "DA", family = "gaussian", datasources = conns)
print(result)

Expected output: coefficients, SE, and p-values matching gt_gaussian, deviance within 0.01% of deviance(gt_gaussian).

3.3 Binomial GLM (low birth weight)

result <- ds.vertGLM(low ~ age + lwt + race + smoke + ptl + ht + ui,
                     data = "DA", family = "binomial", datasources = conns)
print(result)

Expected: canonical deviance within 0.11% of deviance(gt_binomial).

3.4 Poisson GLM (physician visits)

result <- ds.vertGLM(ftv ~ age + lwt + race + smoke + ptl + ht + ui,
                     data = "DA", family = "poisson", datasources = conns)
print(result)

Expected: canonical deviance within 0.18% of deviance(gt_poisson).

3.5 Correlation and PCA

# Correlation matrix across all features from both hospitals
cor_result <- ds.vertCor("DA", datasources = conns)
print(cor_result)

# Compare with centralised
centralised_cor <- cor(merged[, c("age", "lwt", "race", "smoke", "ptl", "ht", "ui")])
max_cor_error <- max(abs(cor_result$correlation - centralised_cor))
# Expected: max element-wise error < 0.001

# PCA from the correlation matrix
pca_result <- ds.vertPCA(cor_result = cor_result)
print(pca_result)

# Compare eigenvalues
centralised_pca <- eigen(centralised_cor)
max_eigen_error <- max(abs(pca_result$eigenvalues - centralised_pca$values))
# Expected: max eigenvalue error < 0.0001
DSI::datashield.logout(conns)

Privacy guarantees

Throughout all analyses, the analyst (client) received only:

  • \(p\)-dimensional gradient sums (aggregates over all \(n\) observations)
  • \(p \times p\) correlation scalars
  • Canonical deviance (1 scalar)
  • Standard errors (\(p\) scalars)

No individual patient data (\(\eta_i\), \(\mu_i\), \(y_i\), \(x_{ij}\)) was disclosed to the analyst or to any other server.
